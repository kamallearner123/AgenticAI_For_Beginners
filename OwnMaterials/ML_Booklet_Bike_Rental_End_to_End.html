Problem Statement & Process

You operate a city bike‑sharing program. You need to predict hourly rental demand (a regression task) and also flag high‑demand hours (a classification task) so you can place bikes optimally, plan maintenance, and trigger surge staffing.

  • Inputs: hour of day, weather (temperature, humidity, wind, precipitation), weekend/holiday flags.
  • Outputs: rentals (count per hour), and is_high_demand (binary).
  • KPIs: RMSE/R² for regression; Accuracy/ROC‑AUC for classification.

Chapter 2

Baselines

Always start with simple baselines to establish reference metrics. If a fancy model can’t beat them, it’s likely overfitting or mis‑specified. 

# Baseline models
from sklearn.dummy import DummyRegressor, DummyClassifier

# Regression baseline: predict the mean
baseline_reg = Pipeline(steps=[("prep", preprocessor), ("mdl", DummyRegressor(strategy="mean"))])
baseline_reg.fit(X_train_r, y_train_r)
pred_r = baseline_reg.predict(X_test_r)
print("Baseline (Reg) RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

# Classification baseline: most frequent class
baseline_clf = Pipeline(steps=[("prep", preprocessor), ("mdl", DummyClassifier(strategy="most_frequent"))])
baseline_clf.fit(X_train_c, y_train_c)
pred_c = baseline_clf.predict(X_test_c)
print("Baseline (Clf) ACC:", accuracy_score(y_test_c, pred_c))
Chapter 3

Linear & Logistic Regression

Linear Regression gives a fast, interpretable baseline for counts; Logistic Regression is a strong start for classifications, especially with standardized inputs. 

# Setup
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.metrics import mean_squared_error, r2_score, accuracy_score, roc_auc_score, classification_report

# Assuming you already loaded the dataset as df (like the one in this page)
features = ["hour","temp_c","humidity","wind_kph","is_weekend","is_holiday","precip_mm"]
target_reg = "rentals"
target_clf = "is_high_demand"

X = df[features]
y_reg = df[target_reg]
y_clf = df[target_clf]

num_cols = ["hour","temp_c","humidity","wind_kph","precip_mm"]
cat_cols = ["is_weekend","is_holiday"]

preprocessor = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(drop="if_binary"), cat_cols),
    ]
)

X_train_r, X_test_r, y_train_r, y_test_r = train_test_split(X, y_reg, test_size=0.25, random_state=42, stratify=None)
X_train_c, X_test_c, y_train_c, y_test_c = train_test_split(X, y_clf, test_size=0.25, random_state=42, stratify=y_clf)

# Linear Regression & Logistic Regression
from sklearn.linear_model import LinearRegression, LogisticRegression

linreg = Pipeline(steps=[("prep", preprocessor), ("mdl", LinearRegression())])
linreg.fit(X_train_r, y_train_r)
pred_r = linreg.predict(X_test_r)
print("Linear Regression RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))
print("Linear Regression R^2 :", r2_score(y_test_r, pred_r))

logreg = Pipeline(steps=[("prep", preprocessor), ("mdl", LogisticRegression(max_iter=200))])
logreg.fit(X_train_c, y_train_c)
pred_c = logreg.predict(X_test_c)
proba_c = logreg.predict_proba(X_test_c)[:,1]
print("Logistic Regression ACC   :", accuracy_score(y_test_c, pred_c))
print("Logistic Regression ROC-AUC:", roc_auc_score(y_test_c, proba_c))

Chapter 5

Ensembles: Random Forest & Gradient Boosting

Ensembles combine many weak learners into a strong learner. In many tabular problems, these are competitive with state‑of‑the‑art. 

# Ensembles: Random Forest & Gradient Boosting
from sklearn.ensemble import RandomForestRegressor, RandomForestClassifier, GradientBoostingRegressor, GradientBoostingClassifier

rf_reg = Pipeline(steps=[("prep", preprocessor), ("mdl", RandomForestRegressor(n_estimators=200, random_state=42))])
rf_reg.fit(X_train_r, y_train_r)
pred_r = rf_reg.predict(X_test_r)
print("Random Forest (Reg) RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

rf_clf = Pipeline(steps=[("prep", preprocessor), ("mdl", RandomForestClassifier(n_estimators=300, random_state=42))])
rf_clf.fit(X_train_c, y_train_c)
pred_c = rf_clf.predict(X_test_c)

print("Random Forest (Clf) ACC   :", accuracy_score(y_test_c, pred_c))
print("Random Forest (Clf) ROC-AUC:", roc_auc_score(y_test_c, proba_c))

gbr = Pipeline(steps=[("prep", preprocessor), ("mdl", GradientBoostingRegressor(random_state=42))])
gbr.fit(X_train_r, y_train_r)
pred_r = gbr.predict(X_test_r)
print("Gradient Boosting (Reg) RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

gbc = Pipeline(steps=[("prep", preprocessor), ("mdl", GradientBoostingClassifier(random_state=42))])
gbc.fit(X_train_c, y_train_c)
pred_c = gbc.predict(X_test_c)
proba_c = gbc.predict_proba(X_test_c)[:,1]

SVM, k‑NN, and Naive Bayes

These families provide different bias‑variance trade‑offs and decision boundaries. Test them to better understand your data manifold. 

# SVM, k-NN, Naive Bayes
from sklearn.svm import SVR, SVC
from sklearn.neighbors import KNeighborsRegressor, KNeighborsClassifier
from sklearn.naive_bayes import GaussianNB

# SVM (Reg & Clf)
svr = Pipeline(steps=[("prep", preprocessor), ("mdl", SVR())])
svr.fit(X_train_r, y_train_r)
pred_r = svr.predict(X_test_r)
print("SVR RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

svc = Pipeline(steps=[("prep", preprocessor), ("mdl", SVC(probability=True))])
svc.fit(X_train_c, y_train_c)
pred_c = svc.predict(X_test_c)
proba_c = svc.predict_proba(X_test_c)[:,1]
print("SVC ACC   :", accuracy_score(y_test_c, pred_c))
print("SVC ROC-AUC:", roc_auc_score(y_test_c, proba_c))

# k-NN
knn_reg = Pipeline(steps=[("prep", preprocessor), ("mdl", KNeighborsRegressor(n_neighbors=7))])
knn_reg.fit(X_train_r, y_train_r)
pred_r = knn_reg.predict(X_test_r)
print("kNN (Reg) RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

knn_clf = Pipeline(steps=[("prep", preprocessor), ("mdl", KNeighborsClassifier(n_neighbors=9))])
knn_clf.fit(X_train_c, y_train_c)
pred_c = knn_clf.predict(X_test_c)
print("kNN (Clf) ACC:", accuracy_score(y_test_c, pred_c))

# Naive Bayes (classification only; numeric features work with GaussianNB)
# We'll use a different pipeline because GaussianNB does not accept sparse matrices by default.
nb_prep = ColumnTransformer(
    transformers=[("num", StandardScaler(), ["hour","temp_c","humidity","wind_kph","precip_mm"])],
    remainder="drop"
)
nb_clf = Pipeline(steps=[("prep", nb_prep), ("mdl", GaussianNB())])
nb_clf.fit(X_train_c, y_train_c)
pred_c = nb_clf.predict(X_test_c)
print("Naive Bayes (Clf) ACC:", accuracy_score(y_test_c, pred_c))
Chapter 7

Unsupervised Learning (PCA + KMeans)

Even when your goal is prediction, unsupervised steps help visualize structure, detect cohorts, and check for anomalies. 

# Unsupervised: PCA (dimensionality reduction) + KMeans (clustering)
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans

num_only = df[["hour","temp_c","humidity","wind_kph","precip_mm"]]
scaler = StandardScaler()
Z = scaler.fit_transform(num_only)

pca = PCA(n_components=2, random_state=42)
Z2 = pca.fit_transform(Z)
print("Explained variance ratio:", pca.explained_variance_ratio_)

km = KMeans(n_clusters=3, n_init=10, random_state=42)
clusters = km.fit_predict(Z2)
df["cluster_2D"] = clusters

# Quick plot (matplotlib)
import matplotlib.pyplot as plt
plt.figure()
for k in range(3):
    subset = Z2[clusters==k]
    plt.scatter(subset[:,0], subset[:,1], s=12, label=f"cluster {k}")
plt.legend()
plt.title("KMeans clusters in PCA space")
plt.show()
Chapter 8

Model Selection: Cross‑Validation & Hyperparameters

Use cross‑validation to estimate generalization and tune hyperparameters systematically. 

# Model Selection: Cross-validation & simple hyperparameter search
from sklearn.model_selection import GridSearchCV

# Example: RandomForestRegressor param sweep
from sklearn.ensemble import RandomForestRegressor
rf_pipe = Pipeline(steps=[("prep", preprocessor), ("mdl", RandomForestRegressor(random_state=42))])

param_grid = {
    "mdl__n_estimators": [100, 200],
    "mdl__max_depth": [None, 8, 12],
    "mdl__min_samples_split": [2, 5]
}

gcv = GridSearchCV(rf_pipe, param_grid, cv=3, scoring="neg_root_mean_squared_error", n_jobs=-1)
gcv.fit(X_train_r, y_train_r)

print("Best params:", gcv.best_params_)
best_reg = gcv.best_estimator_
pred_r = best_reg.predict(X_test_r)
print("Tuned RF (Reg) RMSE:", mean_squared_error(y_test_r, pred_r, squared=False))

Next Steps & Deployment Tips

  • Persist artifacts: save pipelines with joblib; include preprocessing in the pipeline.
  • Validate data at inference: schema checks, ranges, missing values.
  • Monitor: track input drift, performance decay; set alerts.
  • Retrain: on a cadence or triggered by drift; keep champion/challenger.
  • Explainability: permutation importance, SHAP (when appropriate), partial dependence.
